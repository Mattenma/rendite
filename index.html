    <script>

        function calculate() {

            var kaufpreis = parseFloat(document.getElementById('kaufpreis').value) || 0;

            var miete = parseFloat(document.getElementById('miete').value) || 0;

            var eigenmittel = parseFloat(document.getElementById('eigenmittel').value) || 0;

            var hypothekarzins = parseFloat(document.getElementById('hypothekarzins').value) || 0;

            var verwaltungskosten = parseFloat(document.getElementById('verwaltungskosten').value) || 0;


            // Fremdmittel berechnen

            var fremdmittel = kaufpreis - eigenmittel;

            document.getElementById('fremdmittel').value = fremdmittel ? Math.round(fremdmittel) : '';


            // Formel zur Berechnung der Netto Rendite

            var jahresmiete = miete * 1000 * 12;

            var nettoMiete = jahresmiete * (1 - (verwaltungskosten / 100));

            var zinskosten = fremdmittel * 1000000 * (hypothekarzins / 100);

            var nettoertrag = nettoMiete - zinskosten;

            var nettoRendite = (nettoertrag / (kaufpreis * 1000000)) * 100;


            var nettoRenditeElement = document.getElementById('nettoRendite');

            nettoRenditeElement.innerText = nettoRendite.toFixed(2) + ' %';


            if (nettoRendite < 1) {

                nettoRenditeElement.style.color = 'red';

            } else if (nettoRendite < 2) {

                nettoRenditeElement.style.color = 'orange';

            } else {

                nettoRenditeElement.style.color = 'green';

            }

        }


        window.onload = function() {

            calculate();

        }

    </script>

</head>

<body>

    <div class="container">

        <h1>Rendite Schätzer</h1>

        

        <h2>Netto Rendite: <span id="nettoRendite">0.00 %</span></h2>


        <label for="kaufpreis">Kaufpreis (in Mio.):</label>

        <input type="number" id="kaufpreis" step="0.01" maxlength="4" oninput="calculate()">

        

        <label for="miete">Schätzwert Miete / Mt. (in Tausend):</label>

        <input type="number" id="miete" step="0.01" maxlength="4" oninput="calculate()">

        

        <label for="eigenmittel">Eigenmittel (in Mio.):</label>

        <input type="number" id="eigenmittel" step="0.01" value="1" maxlength="4" oninput="calculate()">

        

        <label for="fremdmittel">Fremdmittel (in Mio.):</label>

        <input type="number" id="fremdmittel" step="0.01" maxlength="4" readonly>

        

        <label for="hypothekarzins">Hypothekarzins (%):</label>

        <input type="number" id="hypothekarzins" step="0.01" value="2.5" maxlength="4" oninput="calculate()">

        

        <label for="verwaltungskosten">Verwaltungs & Betriebskosten (%):</label>

        <input type="number" id="verwaltungskosten" step="0.01" value="20" maxlength="4" oninput="calculate()">
